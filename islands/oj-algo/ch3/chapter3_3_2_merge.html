3.2 еҪ’е№¶жҺ’еәҸ

Merge SortпјҡеҲҶиҖҢжІ»д№Ӣзҡ„йӯ”жі• вңЁ

рҹ§ҷвҖҚвҷӮпёҸ еҲҶжІ»жі•зҡ„еҘҘд№ү

еҒҮи®ҫдҪ жүӢйҮҢжңүдёҖеүҜд№ұеәҸзҡ„жү‘е…ӢзүҢгҖӮеҰӮжһңдҪ жғіжҠҠе®ғд»¬жҺ’еҘҪеәҸпјҢеҪ’е№¶жҺ’еәҸзҡ„йӯ”жі•еёҲдјҡе‘ҠиҜүдҪ дёӨдёӘе’’иҜӯпјҡ

  1. еҲҶиЈӮе’’ (Divide)пјҡ еҲ«з®ЎйӮЈд№ҲеӨҡпјҢе…ҲжҠҠиҝҷе ҶзүҢдёҖеҲҶдёәдәҢпјҢе·Ұиҫ№дёҖе ҶпјҢеҸіиҫ№дёҖе ҶгҖӮеҰӮжһңиҝҳе«ҢеӨҡпјҢе°ұз»§з»ӯеҲҶпјҢзӣҙеҲ°жҜҸе ҶеҸӘжңү1еј зүҢдёәжӯўгҖӮпјҲ1еј зүҢеӨ©з„¶е°ұжҳҜжңүеәҸзҡ„пјҒпјү
  2. иһҚеҗҲе’’ (Conquer/Merge)пјҡ жҠҠдёӨе Ҷе·Із»ҸжңүеәҸзҡ„зүҢпјҢеҗҲе№¶жҲҗдёҖеӨ§е ҶжңүеәҸзҡ„зүҢгҖӮ

дёҖзӣҙйҮҚеӨҚиҝҷдёӘиҝҮзЁӢпјҢжңҖеҗҺжүҖжңүзҡ„зүҢе°ұзҘһеҘҮең°жҺ’еҘҪе•ҰпјҒ

рҹҺЁ е®Ҹи§Ӯеӣҫи§Јпјҡе…ҲеҲҶеҗҺеҗҲ

зңӢзңӢиҝҷдёӘеҖ’зқҖзҡ„вҖңж ‘вҖқпјҢе®ғжҳҜеҰӮдҪ•еӨ„зҗҶ [8, 4, 5, 7] иҝҷз»„ж•°еӯ—зҡ„гҖӮ

дёҖзӣҙеҲҶеҲ°еҸӘеү©1дёӘпјҢз„¶еҗҺдёӨдёӨжҜ”иҫғпјҢжңүеәҸең°еҗҲиө·жқҘпјҒ

рҹ”Қ еҫ®и§Ӯеӣҫи§ЈпјҡеҰӮдҪ•еҗҲе№¶пјҹ(Merge)

иҝҷжҳҜжңҖе…ій”®зҡ„дёҖжӯҘпјҒеҒҮи®ҫжҲ‘д»¬жңүдёӨйҳҹе·Із»ҸжҺ’еҘҪеәҸзҡ„еӯҰз”ҹпјҢиҰҒеҗҲе№¶жҲҗдёҖеӨ§йҳҹгҖӮ

е·Ұйҳҹпјҡ [2, 5, 9]     еҸійҳҹпјҡ [1, 4, 6]

и§„еҲҷпјҡ жҜҸж¬ЎеҸӘзңӢдёӨйҳҹзҡ„жҺ’еӨҙе…өпјҢи°Ғе°Ҹи°Ғе…Ҳиө°пјҢиө°еҲ°ж–°йҳҹдјҚйҮҢеҺ»гҖӮ

иҜҰз»Ҷеӣҫи§ЈжөҒзЁӢ

рҹ’» зј–зЁӢж—¶й—ҙ

еҪ’е№¶жҺ’еәҸзҡ„д»Јз ҒзЁҚеҫ®й•ҝдёҖзӮ№зӮ№пјҢеӣ дёәе®ғйңҖиҰҒдёҖдёӘиҫ…еҠ©ж•°з»„пјҲйўқеӨ–зҡ„з©әй—ҙпјүжқҘжҡӮеӯҳеҗҲе№¶еҗҺзҡ„з»“жһңгҖӮ

#include <iostream> using namespace std; int a[1005]; // еҺҹж•°з»„ int tmp[1005]; // иҫ…еҠ©ж•°з»„пјҢз”ЁжқҘдёҙж—¶еӯҳж”ҫеҗҲе№¶еҘҪзҡ„ж•°жҚ® // еҪ’е№¶жҺ’еәҸеҮҪж•°пјҡеӨ„зҗҶ l еҲ° r иҢғеӣҙ void merge_sort(int l, int r) { // 1. йҖ’еҪ’еҮәеҸЈпјҡеҰӮжһңеҸӘеү©дёҖдёӘж•°пјҢе°ұдёҚз”ЁеҲҶдәҶ if (l >= r) return; // 2. жүҫдёӯй—ҙзӮ№пјҢдёҖеҲҶдёәдәҢ int mid = (l + r) / 2; // 3. йҖ’еҪ’пјҡеҲҶеҲ«еҺ»жҺ’е·Ұиҫ№е’ҢеҸіиҫ№ merge_sort(l, mid); // е·ҰеҚҠиҫ№ merge_sort(mid + 1, r); // еҸіеҚҠиҫ№ // 4. еҗҲе№¶ (Merge)пјҡжңҖж ёеҝғзҡ„йғЁеҲҶпјҒ int i = l; // е·Ұиҫ№зҡ„жҺ’еӨҙе…ө int j = mid + 1; // еҸіиҫ№зҡ„жҺ’еӨҙе…ө int k = 0; // иҫ…еҠ©ж•°з»„зҡ„жҢҮй’Ҳ // еҸӘиҰҒдёӨиҫ№йғҪжңүдәәпјҢе°ұPK while(i <= mid && j <= r) { if(a[i] <= a[j]) { tmp[k++] = a[i++]; // е·Ұиҫ№е°ҸпјҢе·Ұиҫ№ж”ҫе…Ҙtmp } else { tmp[k++] = a[j++]; // еҸіиҫ№е°ҸпјҢеҸіиҫ№ж”ҫе…Ҙtmp } } // еҰӮжһңжңүдёҖиҫ№иҝҳжІЎиө°е®ҢпјҢзӣҙжҺҘе…ЁйғЁжҺҘеңЁеҗҺйқў while(i <= mid) tmp[k++] = a[i++]; while(j <= r) tmp[k++] = a[j++]; // 5. жҗ¬еӣһеҺ»пјҡжҠҠжҺ’еҘҪеәҸзҡ„tmpж•°з»„пјҢеӨҚеҲ¶еӣһеҺҹж•°з»„a for(int x = 0; x < k; x++) { a[l + x] = tmp[x]; } } int main() { int n; cin >> n; for(int i = 0; i < n; i++) cin >> a[i]; merge_sort(0, n - 1); for(int i = 0; i < n; i++) cout << a[i] << " "; return 0; }
def merge_sort(arr): # 1. йҖ’еҪ’еҮәеҸЈпјҡеҰӮжһңеҸӘеү©дёҖдёӘж•°жҲ–з©әпјҢзӣҙжҺҘиҝ”еӣһ if len(arr) <= 1: return arr # 2. жүҫдёӯй—ҙзӮ№пјҢеҲҮзүҮ (PythonеҲҮзүҮеҫҲж–№дҫҝпјҒ) mid = len(arr) // 2 left_half = arr[:mid] right_half = arr[mid:] # 3. йҖ’еҪ’пјҡеҲҶеҲ«еҺ»жҺ’е·Ұиҫ№е’ҢеҸіиҫ№ left_sorted = merge_sort(left_half) right_sorted = merge_sort(right_half) # 4. еҗҲе№¶ (Merge) return merge(left_sorted, right_sorted) def merge(left, right): result = [] i = 0 # е·ҰжҢҮй’Ҳ j = 0 # еҸіжҢҮй’Ҳ # еҸӘиҰҒдёӨиҫ№йғҪжІЎиө°е®ҢпјҢе°ұPK while i < len(left) and j < len(right): if left[i] <= right[j]: result.append(left[i]) i += 1 else: result.append(right[j]) j += 1 # жҠҠеү©дҪҷзҡ„жҺҘеңЁеҗҺйқў result.extend(left[i:]) result.extend(right[j:]) return result # жөӢиҜ• data = [8, 4, 5, 7, 1, 3, 6, 2] sorted_data = merge_sort(data) print("жҺ’еәҸз»“жһң:", sorted_data)

рҹҢҹ дёәд»Җд№Ҳе®ғеҫҲеҺүе®іпјҹ

рҹӣЎпёҸ и¶…зә§зЁіе®ҡ

е®ғзҡ„йҖҹеәҰйқһеёёзЁіе®ҡпјҲO(nlogn)пјүпјҢдёҚз®Ўж•°жҚ®еӨҡд№ұпјҢе®ғйғҪдёҚж…ҢгҖӮиҖҢдё”е®ғжҳҜзЁіе®ҡжҺ’еәҸпјҢд№ҹе°ұжҳҜиҜҙпјҢеҰӮжһңдёӨдёӘж•°еӯ—дёҖж ·еӨ§пјҲжҜ”еҰӮдёӨдёӘ5пјүпјҢжҺ’е®ҢеәҸеҗҺпјҢеүҚйқўзҡ„5иҝҳжҳҜеңЁеүҚйқўгҖӮ

рҹӨқ йҖҶеәҸеҜ№д№ӢзҺӢ

еңЁдҝЎжҒҜеӯҰеҘҘиөӣдёӯпјҢеҰӮжһңдҪ йҒҮеҲ°дёҖйҒ“йўҳеҸ«вҖңжұӮйҖҶеәҸеҜ№вҖқпјҢеҲ«зҠ№иұ«пјҢзӣҙжҺҘеҸ¬е”ӨеҪ’е№¶жҺ’еәҸпјҒиҝҷжҳҜе®ғзҡ„жӢҝжүӢеҘҪжҲҸгҖӮ
